using System;
using System.Collections.Generic;
using System.Linq;

namespace LambdaDemo
{
    public class Student
    {
        public int Id { get; set; }
        public string Name { get; set; }
        public int Age { get; set; }
    }
    public class Employee
    {
        public int Id { get; set; }
        public string FirstName { get; set; }
        public string LastName { get; set; }
    }


    class Program
    {
        static void Main(string[] args)
        {

            List<Student> students = new List<Student>
            {
                new Student { Id = 1, Name = "Suman", Age = 17 },
                new Student { Id = 2, Name = "Arjun", Age = 18 },
                new Student { Id = 3, Name = "Meera", Age = 20 },
                new Student { Id = 4, Name = "Rahul", Age = 16 },
                new Student { Id = 5, Name = "Anjali", Age = 19 }
            };


            var filteredStudents = students.Where(s => s.Age >= 18);

            Console.WriteLine("Students aged 18 or older:\n");

            foreach (var student in filteredStudents)
            {
                Console.WriteLine($"Id: {student.Id}, Name: {student.Name}, Age: {student.Age}");
            }

            Console.WriteLine();

             List<Employee> employees = new List<Employee>
            {
                new Employee { Id = 1, FirstName = "Suman", LastName = "Kumar" },
                new Employee { Id = 2, FirstName = "Arjun", LastName = "Sharma" },
                new Employee { Id = 3, FirstName = "Meera", LastName = "Reddy" },
                new Employee { Id = 4, FirstName = "Rahul", LastName = "Agarwal" },
                new Employee { Id = 5, FirstName = "Anjali", LastName = "Patel" }
            };

            // Sort by LastName in ascending order using lambda
            var sortedEmployees = employees
                                    .OrderBy(e => e.LastName)
                                    .ToList();

            Console.WriteLine("Employees sorted by Last Name (Ascending):\n");

            foreach (var emp in sortedEmployees)
            {
                Console.WriteLine($"Id: {emp.Id}, Name: {emp.FirstName} {emp.LastName}");
            }
            
            Console.WriteLine();
            int[] numbers = { 10, 20, 30, 40, 50 };

            // Use lambda expression inside Sum()
            int totalSum = numbers.Sum(n => n);

            Console.WriteLine("\n\nTotal Sum: " + totalSum);

            Console.WriteLine();

            List<string> names = new List<string>
            {
                "suman",
                "arjun",
                "meera",
                "rahul",
                "anjali"
            };

            var upperCaseNames = names
                                 .Select(name => name.ToUpper())
                                 .ToList();

            Console.WriteLine("Original List:");
            foreach (var name in names)
            {
                Console.WriteLine(name);
            }

            Console.WriteLine("\nTransformed List (Uppercase):");
            foreach (var name in upperCaseNames)
            {
                Console.WriteLine(name);
            }
        }
    }
}

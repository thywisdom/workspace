using System;
using System.Collections.Generic;
using System.Linq;

namespace LinqFilterSortDemo
{
    public class Employee
    {
        public int Id { get; set; }
        public string Name { get; set; }
        public double Salary { get; set; }
    }

    class Program
    {
        static void Main(string[] args)
        {
            // Sample employee data
            List<Employee> employees = new List<Employee>
            {
                new Employee { Id = 1, Name = "John Doe", Salary = 60000 },
                new Employee { Id = 2, Name = "Alice Smith", Salary = 45000 },
                new Employee { Id = 3, Name = "Robert Brown", Salary = 75000 },
                new Employee { Id = 4, Name = "David Wilson", Salary = 50000 },
                new Employee { Id = 5, Name = "Emma Johnson", Salary = 82000 }
            };

            // LINQ: Filter salary > 50000 AND sort by Name ascending
            var result = employees
                         .Where(e => e.Salary > 50000)
                         .OrderBy(e => e.Name)
                         .ToList();

            Console.WriteLine("Employees with Salary > 50,000 (Sorted by Name):\n");

            foreach (var emp in result)
            {
                Console.WriteLine($"Id: {emp.Id}, Name: {emp.Name}, Salary: {emp.Salary}");
            }
        }
    }
}
